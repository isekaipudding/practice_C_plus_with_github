#include <iostream>

using namespace std;

int main() {
    int a, b, c;
    cout << "computer basic score -> ";
    cin >> a;
    cout << "mathematics score -> ";
    cin >> b;
    cout << "english score -> ";
    cin >> c;

    int total = a + b + c;
    double average = (double)total / 3;

    cout << "total score -> " << total << "\n";
    cout << "average score -> " << average;

    return 0;
}
