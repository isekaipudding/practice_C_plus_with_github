#include <iostream>

using namespace std;

int main() {
    int x, y;
    cout << "Input x_axis and y_axis -> ";
    cin >> x >> y;

    if(x > 0 && y > 0) cout << "First Quadrant";
    else if(x < 0 && y > 0) cout << "Second Quadrant";
    else if(x < 0 && y < 0) cout << "Third Quadrant";
    else if(x > 0 && y < 0) cout << "Fourth Quadrant";

    return 0;
}
