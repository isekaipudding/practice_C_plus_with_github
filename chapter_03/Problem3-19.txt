#include <iostream>

using namespace std;

int main() {
    int sum = 0;

    for(int i = 1; i < 4; i++) {
        for(int j = 6 - 2 * i; j > 0; j--) {
            cout << ' ';
        }
        for(int j = 2 * i - 1; j > 0; j--) {
            sum += 1;
            cout << sum << ' ';
        }
        cout << '\n';
    }

    return 0;
}
