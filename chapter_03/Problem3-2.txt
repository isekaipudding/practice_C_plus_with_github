#include <iostream>

using namespace std;

int main() {
    cout << "\n a. \n";
    for(int i = 1; i <= 10; i++) {
        if(i%3 == 0)
        cout << "\t" << i;
    }
    cout << '\n';

    cout << "\n b. \n";
    for(int i = 10; i >= 0; i-=3) {
        cout << "\t" << i;
    }
    cout << '\n';

    cout << "\n c. \n";
    int x, y;
    for(x = 0; x < 3; x++) {
        for(y = 2; y >= 0; y--) {
            cout << "\t" << x << "\t" << y << '\n';
        }
    }
    cout << '\n';

    cout << "\n d. \n";
    for(int i = 0; i <= 10; i+=3) {
        if(i % 5 == 0) {
            cout << "\t" << i;
        }
    }
    cout << '\n';

    return 0;
}
