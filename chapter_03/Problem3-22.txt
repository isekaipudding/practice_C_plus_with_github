#include <iostream>

using namespace std;

int main() {
    double a;
    int n;
    double result = 1.0;
    cout << "Input one Double number -> ";
    cin >> a;
    cout << "Input one Integer number -> ";
    cin >> n;

    for(int i = 0; i < n; i++) {
        result *= a;
    }

    cout << result;

    return 0;
}
