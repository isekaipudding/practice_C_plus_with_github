#include <iostream>

using namespace std;

int main() {
    int n;

    cout << "Input one Integer -> ";
    cin >> n;

    cout << "n" << '\t' << "sum of (from 1 to n)" << '\n';
    cout << "-----------------------------------" << '\n';

    for(int i = 1; i <= n; i++) {
        cout << i << '\t' << (i + 1) * i / 2 << '\n';
    }

    return 0;
}
