#include <iostream>

using namespace std;

int main() {
    int x, y, z;
    char ch;

    cout << "Input x -> ";
    cin >> x;
    if(100 <= x && x < 200) cout << "ok" << '\n';
    cout << "Input one ch -> ";
    cin >> ch;
    if(ch == 'a' || ch == 'A') cout << "ok" << '\n';
    cout << "Input y, z -> ";
    cin >> y >> z;
    if(10 <= y && y <=20 || z == 30) cout << "ok" << '\n';

    return 0;
}
