#include <iostream>

using namespace std;

int main() {
    for(int i = 2; i <= 9; i++) {
        for(int j = 2; j <= 9; j++) {
            cout << i << " x " << j << " = " << i * j << '\t';
        }
        cout << '\n';
    }
    cout << "Press any key for exit -> ";
    int e;
    cin >> e;

    return 0;
}
